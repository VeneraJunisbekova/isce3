101  // Comparison operator for floating point
102  template <typename U = T, std::enable_if_t<!std::is_compound<U>::value, int> = 0>
103  inline bool operator==(const isce::core::LUT2d<T> & other) const {
104  // Check coordinates and dimensions first
105  bool equal = _data.width() == other.width();
106  equal *= _data.length() == other.length();
107  if (!equal) {
108  return false;
109  }
110  equal *= isce::core::compareFloatingPoint(_xstart, other.xStart());
111  equal *= isce::core::compareFloatingPoint(_ystart, other.yStart());
112  equal *= isce::core::compareFloatingPoint(_dx, other.xSpacing());
113  equal *= isce::core::compareFloatingPoint(_dy, other.ySpacing());
114  if (!equal) {
115  return false;
116  }
117  // If we made it this far, check contents
118  const isce::core::Matrix<T> & otherMat = other.data();
119  for (size_t i = 0; i < _data.length(); ++i) {
120  for (size_t j = 0; j < _data.width(); ++j) {
121  equal *= isce::core::compareFloatingPoint(_data(i,j), otherMat(i,j));
122  }
123  }
124  return equal;
125  }
