78  /* Inline function for input checking on vector lengths (primarily to check to see if 3D vector
79  * has the correct number of inputs, but is generalized to any length). 'vec_name' is passed in
80  * by the wrapper macro (stringified vector name), and 'parent_func' is passed in the same way
81  * through __PRETTY_FUNCTION__*/
82  template<typename T>
83  inline void checkVecLenDebug(const std::vector<T> &vec, size_t len, const char *vec_name,
84  const char *parent_func) {
85  if (vec.size() != len) {
86  std::string errstr = "In '" + std::string(parent_func) + "': Vector '" +
87  std::string(vec_name) + "' has size " +
88  std::to_string(vec.size()) + ", expected size " +
89  std::to_string(len) + ".";
90  throw std::invalid_argument(errstr);
91  }
92  }

114  // Instantiate the vector
115  std::vector<T> data;
116  // Set the first value

131  template<typename T>
132  std::vector<T> linspace(T low, T high, std::size_t number) {
133  // Instantiate the vector
134  std::vector<T> data;
135  // Compute the increment
136  T increment = (high - low) / (number - 1);
137  // Handle cases where number in (0, 1)
138  if (number == 0) {
139  return data;
140  }
141  if (number == 1) {
142  data.push_back(low);
143  return data;
144  }
145  // Loop over the increments and add to vector
146  for (std::size_t i = 0; i < number - 1; ++i) {
147  data.push_back(low + i * increment);
148  }
149  // Add the last element
150  data.push_back(high);
151  // done
152  return data;
153  }

160  template<typename T>
161  std::vector<T> logspace(T first, T last, std::size_t number, T base=10) {
162  // Instantiate the vector
163  std::vector<T> data;
164  // Compute the increment of the powers
165  T increment = (last - first) / (number - 1);
166  // Handle cases where number in (0, 1)
167  if (number == 0) {
168  return data;
169  }
170  if (number == 1) {
171  data.push_back(pow(base, first));
172  return data;
173  }
174  // Loop over the increments and add to vector
175  for (std::size_t i = 0; i < number - 1; ++i) {
176  T value = pow(base, first + i * increment);
177  data.push_back(value);
178  }
179  // Add the last element
180  data.push_back(pow(base, last));
181  // done
182  return data;
183  }
