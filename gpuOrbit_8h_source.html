55  CUDA_DEV inline void gpuOrbit::getStateVector(

57  ) const {
58  // Note we can't really do much in the way of bounds-checking since we can't use the
59  // <stdexcept> library, this is best we have
60  bool valid = !((idx < 0) || (idx >= nVectors));
61  t = (valid ? UTCtime[idx] : 0.);
62  for (int i=0; i<3; i++) {
63  pos[i] = (valid ? position[3*idx+i] : 0.);
