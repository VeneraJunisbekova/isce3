139 bool isce::core::Orbit::
140 operator==(const Orbit & rhs) const {
141  // Some easy checks first
142  bool equal = nVectors == rhs.nVectors;
143  equal *= refEpoch == rhs.refEpoch;
144  if (!equal) {
145  return false;
146  }
147  // If we pass the easy checks, check the orbit contents
148  for (size_t i = 0; i < nVectors; ++i) {
149  equal *= isce::core::compareFloatingPoint(UTCtime[i], rhs.UTCtime[i]);
150  for (size_t j = 0; j < 3; ++j) {
151  equal *= isce::core::compareFloatingPoint(position[3*i+j], rhs.position[3*i+j]);
152  equal *= isce::core::compareFloatingPoint(velocity[3*i+j], rhs.velocity[3*i+j]);
153  }
154  }
155  return equal;
156 }

241 void isce::core::Orbit::
242 addStateVector(double t, const cartesian_t & pos, const cartesian_t & vel) {
243  int vec_idx = 0;
244  while ((vec_idx < nVectors) && (t > UTCtime[vec_idx])) vec_idx++;
245  UTCtime.insert(UTCtime.begin()+vec_idx, t);
246  epochs.insert(epochs.begin()+vec_idx, refEpoch + t);
247  position.insert(position.begin()+(3*vec_idx), pos.begin(), pos.end());
248  velocity.insert(velocity.begin()+(3*vec_idx), vel.begin(), vel.end());
249  nVectors++;
250 }
