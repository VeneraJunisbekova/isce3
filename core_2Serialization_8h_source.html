114  inline void loadFromH5(isce::io::IGroup & group, Orbit & orbit) {
115  // Reset orbit data
116  orbit.position.clear();
117  orbit.velocity.clear();
118  orbit.UTCtime.clear();
119  orbit.epochs.clear();
120 
121  // Load position
122  isce::io::loadFromH5(group, "position", orbit.position);
123 
124  // Load velocity
125  isce::io::loadFromH5(group, "velocity", orbit.velocity);
126 
127  // Load time
128  isce::io::loadFromH5(group, "time", orbit.UTCtime);
129 
130  // Get the reference epoch
131  orbit.refEpoch = isce::io::getRefEpoch(group, "time");
132 
133  // Convert UTC seconds since epoch to timestamp
134  orbit.nVectors = orbit.UTCtime.size();
135  orbit.epochs.resize(orbit.nVectors);
136  for (size_t i = 0; i < orbit.nVectors; ++i) {
137  DateTime date = orbit.refEpoch;
138  date += orbit.UTCtime[i];
139  orbit.epochs[i] = date;
140  }

152  inline void loadFromH5(isce::io::IGroup & group, EulerAngles & euler) {
153 
154  // Create temporary data
155  std::vector<double> time, angles, yaw, pitch, roll;
156  isce::core::DateTime refEpoch;
157 
158  // Load angles
159  isce::io::loadFromH5(group, "eulerAngles", angles);
160 
161  // Load time
162  isce::io::loadFromH5(group, "time", time);
163 
164  // Get the reference epoch
165  refEpoch = isce::io::getRefEpoch(group, "time");
166 
167  // Unpack the angles
168  const double rad = M_PI / 180.0;
169  yaw.resize(time.size());
170  pitch.resize(time.size());
171  roll.resize(time.size());
172  for (size_t i = 0; i < time.size(); ++i) {
173  yaw[i] = rad * angles[i*3 + 0];
174  pitch[i] = rad * angles[i*3 + 1];
175  roll[i] = rad * angles[i*3 + 2];
176  }
177 
178  // Save to EulerAngles object
179  euler.data(time, yaw, pitch, roll);
180  euler.refEpoch(refEpoch);
181  }

304  values.assign(std::begin(v_values), std::end(v_values));
305  // Archive
306  archive(cereal::make_nvp("Coords", coords),
