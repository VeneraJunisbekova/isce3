114  inline void loadFromH5(isce::io::IGroup & group,
115  Orbit & orbit,
116  std::string orbit_type="POE",
117  DateTime refEpoch=MIN_DATE_TIME) {
118  // Reset orbit data
119  orbit.position.clear();
120  orbit.velocity.clear();
121  orbit.UTCtime.clear();
122  orbit.epochs.clear();
123 
124  // Save the reference epoch
125  orbit.refEpoch = refEpoch;
126 
127  // Open subgroup based on the orbit type
128  isce::io::IGroup orbGroup = group.openGroup(orbit_type);
129 
130  // Load position
131  isce::io::loadFromH5(orbGroup, "position", orbit.position);
132 
133  // Load velocity
134  isce::io::loadFromH5(orbGroup, "velocity", orbit.velocity);
135 
136  // Load timestamp
137  std::vector<isce::core::FixedString> timestamps;
138  isce::io::loadFromH5(orbGroup, "timestamp", timestamps);
139  orbit.nVectors = timestamps.size();
140  orbit.UTCtime.resize(orbit.nVectors);
141  orbit.epochs.resize(orbit.nVectors);
142 
143  // Finally, convert timestamps seconds
144  for (int i = 0; i < orbit.nVectors; ++i) {
145  // Make a DateTime and save it
146  DateTime date(std::string(timestamps[i].str));
147  orbit.epochs[i] = date;
148  // Convert to seconds since epoch
149  orbit.UTCtime[i] = date.secondsSinceEpoch(refEpoch);
150  }
151  }

202  inline void loadFromH5(isce::io::IH5File & file, Metadata & meta,
203  std::string mode="primary") {
204 
205  double pri, bandwidth, centerFrequency;
206  std::string sensingStart, lookDir;
207 
208  // Get group for image mode
209  std::string path = "/science/complex_imagery/" + mode + "_mode";
210  isce::io::IGroup imGroup = file.openGroup(path);
211 
212  // Get dimension of imagery
213  std::vector<int> dims = isce::io::getImageDims(imGroup, "hh");
214  meta.length = dims[0];
215  meta.width = dims[1];
216 
217  // Load values
218  isce::io::loadFromH5(imGroup, "slant_range_start", meta.rangeFirstSample);
219  isce::io::loadFromH5(imGroup, "slant_range_spacing", meta.slantRangePixelSpacing);
220  isce::io::loadFromH5(imGroup, "az_time_interval", pri);
221  isce::io::loadFromH5(imGroup, "bandwidth", bandwidth);
222  isce::io::loadFromH5(imGroup, "freq_center", centerFrequency);
223  isce::io::loadFromH5(imGroup, "zero_doppler_start_az_time", sensingStart);
224  isce::io::loadFromH5(file, "/science/metadata/identification/look_direction", lookDir);
225 
226  // Fields not currently defined in HDF5 product
227  meta.numberRangeLooks = 1;
228  meta.numberAzimuthLooks = 1;
229 
230  // Finalize rest of metadata items
231  meta.prf = 1.0 / pri;
232  meta.radarWavelength = SPEED_OF_LIGHT / centerFrequency;
233  meta.sensingStart = sensingStart;
234  if (lookDir.compare("right") == 0) {
235  meta.lookSide = -1;
236  } else if (lookDir.compare("left") == 0) {
237  meta.lookSide = 1;
238  }
239  }

281  template <class Archive, typename T>
282  inline void save(Archive & archive, LUT1d<T> const & lut) {
283  // Copy LUT data from valarrays to vectors
284  std::vector<double> coords(lut.size());
285  std::vector<T> values(lut.size());
286  auto v_coords = lut.coords();
287  auto v_values = lut.values();
288  coords.assign(std::begin(v_coords), std::end(v_coords));
289  values.assign(std::begin(v_values), std::end(v_values));
290  // Archive
291  archive(cereal::make_nvp("Coords", coords),
292  cereal::make_nvp("Values", values));
293  }
