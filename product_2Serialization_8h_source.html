101  inline void loadFromH5(isce::io::IGroup & group, ImageMode & mode,
102  const std::string & modeType) {
103 
104  // Get subgroup for specific image mode
105  isce::io::IGroup modeGroup = group.openGroup(modeType + "_mode");
106  // Set mode type
107  mode.modeType(modeType);
108 
109  // Set the image dimensions
110  std::vector<int> dims = isce::io::getImageDims(modeGroup, "hh");
111  std::array<size_t, 2> arrayDims{static_cast<size_t>(dims[0]),
112  static_cast<size_t>(dims[1])};
113  mode.dataDimensions(arrayDims);
114 
115  // For now, hardcode azimuth and range looks until they exist in the product
116  mode.numberAzimuthLooks(1);
117  mode.numberRangeLooks(1);
118 
119  // Set PRF
120  double value;
121  isce::io::loadFromH5(modeGroup, "az_time_interval", value);
122  mode.prf(1.0 / value);
123 
124  // Set bandwidth
125  isce::io::loadFromH5(modeGroup, "bandwidth", value);
126  mode.rangeBandwidth(value);
127 
128  // Set starting slant range
129  isce::io::loadFromH5(modeGroup, "slant_range_start", value);
130  mode.startingRange(value);
131 
132  // Set slant range pixel spacing
133  isce::io::loadFromH5(modeGroup, "slant_range_spacing", value);
134  mode.rangePixelSpacing(value);
135 
136  // Set wavelength
137  isce::io::loadFromH5(modeGroup, "freq_center", value);
138  mode.wavelength(isce::core::SPEED_OF_LIGHT / value);
139 
140  // Load zero doppler starting azimuth time
141  isce::core::FixedString datestr;
142  isce::io::loadFromH5(modeGroup, "zero_doppler_start_az_time", datestr);
143  // Convert to datetime
144  isce::core::DateTime date(std::string(datestr.str));
145  // Set
146  mode.startAzTime(date);
147 
148  // Load zero doppler ending azimuth time
149  isce::io::loadFromH5(modeGroup, "zero_doppler_end_az_time", datestr);
150  // Convert to datetime
151  date = std::string(std::string(datestr.str));
152  // Set
153  mode.endAzTime(date);
154  }
