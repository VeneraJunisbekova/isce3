101 T isce::core::LUT1d<T>::
102 eval(double x) const {
103 
104  // Check bounds to see if we need to perform linear extrapolation
105  const int n = _coords.size();
106  if (x < _coords[0]) {
107  if (_extrapolate) {
108  const double dx = _coords[0] - _coords[1];
109  const double dy = _values[0] - _values[1];
110  const double d = x - _coords[1];
111  T result = (dy / dx) * d + _values[1];
112  return result;
113  } else {
114  pyre::journal::error_t errorChannel("isce.core.LUT1d");
115  errorChannel
116  << pyre::journal::at(__HERE__)
117  << "Out of bounds evaluation for LUT1d."
118  << pyre::journal::newline
119  << pyre::journal::endl;
120  return 0;
121  }
122  } else if (x > _coords[n-1]) {
123  if (_extrapolate) {
124  const double dx = _coords[n-1] - _coords[n-2];
125  const double dy = _values[n-1] - _values[n-2];
126  const double d = x - _coords[n-2];
127  T result = (dy / dx) * d + _values[n-2];
128  return result;
129  } else {
130  pyre::journal::error_t errorChannel("isce.core.LUT1d");
131  errorChannel
132  << pyre::journal::at(__HERE__)
133  << "Out of bounds evaluation for LUT1d."
134  << pyre::journal::newline
135  << pyre::journal::endl;
136  return 0;
137  }
138  }
139 
140  // Otherwise, proceed with interpolation
141  // Iterate over coordinates to find x bounds
142  double xdiff = -100.0;
143  int j;
144  for (j = 0; j < n - 1; ++j) {
145  // Compute difference with current coordinate
146  xdiff = _coords[j] - x;
147  // Break if sign has changed
148  if (xdiff > 0.0)
149  break;
150  }
151 
152  // The indices of the x bounds
153  const int j0 = j - 1;
154  const int j1 = j;
155 
156  // Get coordinates at bounds
157  double x1 = _coords[j0];
158  double x2 = _coords[j1];
159 
160  // Interpolate
161  T result = (x2 - x) / (x2 - x1) * _values[j0] + (x - x1) / (x2 - x1) * _values[j1];
162  return result;
163 }
