193 CUDA_HOSTDEV inline void isce::core::Ellipsoid::
194 lonLatToXyz(const cartesian_t & llh, cartesian_t & xyz) const {
195  /*
196  * Given a lat, lon, and height, produces a geocentric vector.
197  */
198 
199  // Radius of Earth in East direction
200  auto re = rEast(llh[1]);
201  // Parametric representation of a circle as a function of longitude
202  xyz[0] = (re + llh[2]) * std::cos(llh[1]) * std::cos(llh[0]);
203  xyz[1] = (re + llh[2]) * std::cos(llh[1]) * std::sin(llh[0]);
204  // Parametric representation with the radius adjusted for eccentricity
205  xyz[2] = ((re * (1.0 - _e2)) + llh[2]) * std::sin(llh[1]);
206 }

212 CUDA_HOSTDEV inline void isce::core::Ellipsoid::
213 xyzToLonLat(const cartesian_t & xyz, cartesian_t & llh) const {
214  /*
215  * Given a geocentric XYZ, produces a lat, lon, and height above the reference ellipsoid.
216  * VERMEILLE IMPLEMENTATION
217  */
218  // Pre-compute some values
219  const double e4 = _e2 * _e2;
220  const double a2 = _a * _a;
221  // Lateral distance normalized by the major axis
222  double p = (std::pow(xyz[0], 2) + std::pow(xyz[1], 2)) / a2;
223  // Polar distance normalized by the minor axis
224  double q = ((1. - _e2) * std::pow(xyz[2], 2)) / a2;
225  double r = (p + q - e4) / 6.;
226  double s = (e4 * p * q) / (4. * std::pow(r, 3));
227  double t = std::pow(1. + s + std::sqrt(s * (2. + s)), (1./3.));
228  double u = r * (1. + t + (1. / t));
229  double rv = std::sqrt(std::pow(u, 2) + (e4 * q));
230  double w = (_e2 * (u + rv - q)) / (2. * rv);
231  double k = std::sqrt(u + rv + std::pow(w, 2)) - w;
232  // Radius adjusted for eccentricity
233  double d = (k * std::sqrt(std::pow(xyz[0], 2) + std::pow(xyz[1], 2))) / (k + _e2);
234  // Latitude is a function of z and radius
235  llh[1] = std::atan2(xyz[2], d);
236  // Longitude is a function of x and y
237  llh[0] = std::atan2(xyz[1], xyz[0]);
238  // Height is a function of location and radius
239  llh[2] = ((k + _e2 - 1.) * sqrt(std::pow(d, 2) + std::pow(xyz[2], 2))) / k;
240 }
