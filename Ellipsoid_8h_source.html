111 double isce::core::Ellipsoid::rEast(double lat) const {
112  // Radius of Ellipsoid in East direction (assuming latitude-wise symmetry)
113  return _a / std::sqrt(1.0 - (_e2 * std::pow(std::sin(lat), 2)));

120 double isce::core::Ellipsoid::rNorth(double lat) const {
121  // Radius of Ellipsoid in North direction (assuming latitude-wise symmetry)
122  return (_a * (1.0 - _e2)) / std::pow((1.0 - (_e2 * std::pow(std::sin(lat), 2))), 1.5);

130 double isce::core::Ellipsoid::rDir(double hdg, double lat) const {
131  auto re = rEast(lat);
132  auto rn = rNorth(lat);
133  return (re * rn) / ((re * std::pow(std::cos(hdg), 2))
