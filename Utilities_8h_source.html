83  template <typename T>
84  inline std::vector<T> linspace(T low, T high, std::size_t number) {
85  // Instantiate the vector
86  std::vector<T> data;
87  // Compute the increment
88  T increment = (high - low) / (number - 1);
89  // Handle cases where number in (0, 1)
90  if (number == 0) {
91  return data;
92  }
93  if (number == 1) {
94  data.push_back(low);
95  return data;
96  }
97  // Loop over the increments and add to vector
98  for (std::size_t i = 0; i < number - 1; ++i) {
99  data.push_back(low + i * increment);
100  }
101  // Add the last element
102  data.push_back(high);
103  // done
104  return data;
105  }

112  template <typename T>
113  inline std::vector<T> logspace(T first, T last, std::size_t number, T base=10) {
114  // Instantiate the vector
115  std::vector<T> data;
116  // Compute the increment of the powers
117  T increment = (last - first) / (number - 1);
118  // Handle cases where number in (0, 1)
119  if (number == 0) {
120  return data;
121  }
122  if (number == 1) {
123  data.push_back(pow(base, first));
124  return data;
125  }
126  // Loop over the increments and add to vector
127  for (std::size_t i = 0; i < number - 1; ++i) {
128  T value = pow(base, first + i * increment);
129  data.push_back(value);
130  }
131  // Add the last element
132  data.push_back(pow(base, last));
133  // done
134  return data;
135  }

204  compareFloatingPoint(T first, T second) {
205  const T scale = std::max({static_cast<T>(1.0), std::abs(first), std::abs(second)});
