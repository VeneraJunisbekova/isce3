85  inline std::vector<T> linspace(T low, T high, std::size_t number) {
86  // Instantiate the vector
87  std::vector<T> data;
88  // Compute the increment
89  T increment = (high - low) / (number - 1);
90  // Handle cases where number in (0, 1)
91  if (number == 0) {
92  return data;
93  }
94  if (number == 1) {
95  data.push_back(low);
96  return data;
97  }
98  // Loop over the increments and add to vector
99  for (std::size_t i = 0; i < number - 1; ++i) {
100  data.push_back(low + i * increment);
101  }
102  // Add the last element
103  data.push_back(high);
104  // done
105  return data;
106  }

162  inline std::vector<T> logspace(T first, T last, std::size_t number, T base=10) {
163  // Instantiate the vector
164  std::vector<T> data;
165  // Compute the increment of the powers
166  T increment = (last - first) / (number - 1);
167  // Handle cases where number in (0, 1)
168  if (number == 0) {
169  return data;
170  }
171  if (number == 1) {
172  data.push_back(pow(base, first));
173  return data;
174  }
175  // Loop over the increments and add to vector
176  for (std::size_t i = 0; i < number - 1; ++i) {
177  T value = pow(base, first + i * increment);
178  data.push_back(value);
179  }
180  // Add the last element
181  data.push_back(pow(base, last));
182  // done
183  return data;
184  }

344  inline int binarySearch(const std::valarray<double> & array, double value) {
345 
346  // Do the binary search
347  int left = 0;
348  int right = array.size() - 1;
349  int index;
350  while (left <= right) {
351  const int middle = static_cast<int>(std::round(0.5 * (left + right)));
352  if (left == (right - 1)) {
353  index = left;
354  return index;
355  }
356  if (array[middle] <= value) {
357  left = middle;
358  } else if (array[middle] > value) {
359  right = middle;
360  }
361  }
362  index = left;
363  return index;
364  }
