80  template <typename T>
81  inline std::vector<T> linspace(T low, T high, std::size_t number) {
82  // Instantiate the vector
83  std::vector<T> data;
84  // Compute the increment
85  T increment = (high - low) / (number - 1);
86  // Handle cases where number in (0, 1)
87  if (number == 0) {
88  return data;
89  }
90  if (number == 1) {
91  data.push_back(low);
92  return data;
93  }
94  // Loop over the increments and add to vector
95  for (std::size_t i = 0; i < number - 1; ++i) {
96  data.push_back(low + i * increment);
97  }
98  // Add the last element
99  data.push_back(high);
100  // done
101  return data;
102  }

109  template <typename T>
110  inline std::vector<T> logspace(T first, T last, std::size_t number, T base=10) {
111  // Instantiate the vector
112  std::vector<T> data;
113  // Compute the increment of the powers
114  T increment = (last - first) / (number - 1);
115  // Handle cases where number in (0, 1)
116  if (number == 0) {
117  return data;
118  }
119  if (number == 1) {
120  data.push_back(pow(base, first));
121  return data;
122  }
123  // Loop over the increments and add to vector
124  for (std::size_t i = 0; i < number - 1; ++i) {
125  T value = pow(base, first + i * increment);
126  data.push_back(value);
127  }
128  // Add the last element
129  data.push_back(pow(base, last));
130  // done
131  return data;
132  }
