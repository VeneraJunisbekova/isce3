203  template <typename U>
204  inline Interpolator<U> * createInterpolator(
205  isce::core::dataInterpMethod method, size_t order = 6,
206  int sincLen = isce::core::SINC_LEN, int sincSub = isce::core::SINC_SUB
207  ) {
208  if (method == isce::core::BILINEAR_METHOD) {
209  return new isce::core::BilinearInterpolator<U>();
210  } else if (method == isce::core::BICUBIC_METHOD) {
211  return new isce::core::BicubicInterpolator<U>();
212  } else if (method == isce::core::BIQUINTIC_METHOD) {
213  return new isce::core::Spline2dInterpolator<U>(order);
214  } else if (method == isce::core::NEAREST_METHOD) {
215  return new isce::core::NearestNeighborInterpolator<U>();
216  } else if (method == isce::core::SINC_METHOD) {
217  return new isce::core::Sinc2dInterpolator<U>(
218  sincLen, sincSub
219  );
220  } else {
221  return new isce::core::BilinearInterpolator<U>();
222  }
223  }
