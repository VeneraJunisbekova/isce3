222  template <typename U>
223  inline Interpolator<U> * createInterpolator(
224  isce::core::dataInterpMethod method, size_t order = 6,
225  int sincLen = isce::core::SINC_LEN, int sincSub = isce::core::SINC_SUB
226  ) {
227  if (method == isce::core::BILINEAR_METHOD) {
228  return new isce::core::BilinearInterpolator<U>();
229  } else if (method == isce::core::BICUBIC_METHOD) {
230  return new isce::core::BicubicInterpolator<U>();
231  } else if (method == isce::core::BIQUINTIC_METHOD) {
232  return new isce::core::Spline2dInterpolator<U>(order);
233  } else if (method == isce::core::NEAREST_METHOD) {
234  return new isce::core::NearestNeighborInterpolator<U>();
235  } else if (method == isce::core::SINC_METHOD) {
236  return new isce::core::Sinc2dInterpolator<U>(
237  sincLen, sincSub
238  );
239  } else {
240  return new isce::core::BilinearInterpolator<U>();
241  }
242  }
