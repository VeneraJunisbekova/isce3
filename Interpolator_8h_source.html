212 template <typename U>
213 inline Interpolator<U> * createInterpolator(
214  isce::core::dataInterpMethod method, size_t order = 6,
215  int sincLen = isce::core::SINC_LEN, int sincSub = isce::core::SINC_SUB
216  ) {
217  if (method == isce::core::BILINEAR_METHOD) {
218  return new isce::core::BilinearInterpolator<U>();
219  } else if (method == isce::core::BICUBIC_METHOD) {
220  return new isce::core::BicubicInterpolator<U>();
221  } else if (method == isce::core::BIQUINTIC_METHOD) {
222  return new isce::core::Spline2dInterpolator<U>(order);
223  } else if (method == isce::core::NEAREST_METHOD) {
224  return new isce::core::NearestNeighborInterpolator<U>();
225  } else if (method == isce::core::SINC_METHOD) {
226  return new isce::core::Sinc2dInterpolator<U>(
227  sincLen, sincSub
228  );
229  } else {
230  return new isce::core::BilinearInterpolator<U>();
231  }
232 }
